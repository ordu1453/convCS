14void piInit(PIController_t* pi, float kp, float ki, float dt, float outMin, float outMax)
15{
16 pi->kp = kp;
17 pi->ki = ki;
18 pi->dt = dt;
19 pi->outMin = outMin;
20 pi->outMax = outMax;
21 pi->integral = 0.0f;
22 pi->output = 0u;
23}

25unsigned int piUpdate(PIController_t* pi, float setpoint, float measurement)
26{
27 float error = setpoint - measurement;
28 pi->integral += error * pi->ki * pi->dt;
29
30 // Антивиндап
31 if (pi->integral > pi->outMax) pi->integral = pi->outMax;
32 else if (pi->integral < pi->outMin) pi->integral = pi->outMin;
33
34 unsigned int output = pi->kp * error + pi->integral;
35
36 // Ограничение выхода
37 if (output > pi->outMax) output = pi->outMax;
38 else if (output < pi->outMin) output = pi->outMin;
39
40 pi->output = output;
41 return output;
42}

63unsigned int pi2Update(PI2Controller_t* pi2,
64 float voltageSet, float voltageMeas,

69
