23float kalmanFilter(KalmanFilter* kf, float measurement) {
24 if (kf == NULL) return measurement;
25
26 // Этап предсказания
27 // x = x (состояние не меняется в простой модели)
28 kf->p = kf->p + kf->q; // Увеличиваем неопределенность
29
30 // Этап обновления
31 kf->k = kf->p / (kf->p + kf->r); // Вычисляем коэффициент усиления Калмана
32 kf->x = kf->x + kf->k * (measurement - kf->x); // Обновляем оценку
33 kf->p = (1.0f - kf->k) * kf->p; // Обновляем ковариацию ошибки
34
35 kf->last_x = kf->x; // Сохраняем последнее значение
36 return kf->x;
37}

39void kalmanFilterArray(KalmanFilter* kf, float* input, float* output, uint32_t size) {
